try-catch-finally正确关闭connection写法：
finally {
	if (rs != null) {
		try {
			rs.close();
		} catch (SQLException e) {
		}
	}
	if (stmt != null) {
		try {
			stmt.close();
		} catch (SQLException e) {
		}
	}
	if (conn != null) {
		try {
			conn.close();
		} catch (SQLException e) {
		}
	}
}
